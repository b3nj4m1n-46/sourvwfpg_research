% ====================================================================
% 2.  What is Workforce Pell?
% ====================================================================
\frametitleEyebrow{Federal / Workforce Pell Grant}
\begin{frame}{A new Pell category for short, high-quality, employment-aligned programs}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.50\textwidth}
      \rmfamily\fontsize{10pt}{13pt}\selectfont
      \color{textPrimary}
      \eyebrow{program shape}
      \vspace{2pt}
      \begin{itemize}\itemsep1pt
        \item \textbf{8\textendash15 weeks}, 150\textendash599 clock hours
        \item No correspondence, noncredit, or remedial coursework
        \item Must be offered by an \textbf{accredited institution}
        \item Must award academic credit toward an existing
              certificate or degree
      \end{itemize}
      \vspace{8pt}
      \eyebrow{three annual accountability tests}
      \vspace{2pt}
      \begin{itemize}\itemsep1pt
        \item \textbf{70\%} of participants complete within
              150\% of normal time
        \item \textbf{70\%} of completers employed in the
              2nd quarter after exit
        \item Tuition + fees $\leq$ median completer earnings
              $-$ 150\% of poverty line
              ({\color{textMuted}value-added earnings; $\approx$\$23{,}475 in 2026})
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \rmfamily\fontsize{10pt}{13pt}\selectfont
      \color{textPrimary}
      \eyebrow{key timeline}
      \vspace{2pt}
      \begin{itemize}\itemsep1pt
        \item \textbf{Jul 4, 2025} \textemdash{} OBBBA \S 83002 signed
        \item \textbf{Mar 9, 2026} \textemdash{} NPRM published
        \item \textbf{May 19, 2026} \textemdash{} Final rule published
        \item \textbf{Jul 20, 2026} \textemdash{} Rule effective
              ({\color{textMuted}early opt-in Jul 1})
        \item \textbf{AY 2026/27 \textendash{} 2028/29} \textemdash{} Interim phase: only 70/70 applies
        \item \textbf{AY 2029/30} \textemdash{} Transition: first value-added earnings calcs
        \item \textbf{AY 2030/31$+$} \textemdash{} Full accountability
      \end{itemize}
      \vspace{6pt}
      \eyebrow{provisions that favor universities}
      \vspace{2pt}
      \begin{itemize}\itemsep1pt
        \item \textbf{Bachelor's-degree holders are eligible}
        \item \textbf{Employers may deliver up to 25\%} of instruction
              ({\color{textMuted}49\% via Registered Apprenticeship})
        \item \textbf{Governor + state Workforce Board approve first}, then U.S.\ Secretary of Education
      \end{itemize}
    \end{column}
  \end{columns}
  \vspace{4pt}
  {\centering
   \monoLabel{Sources: OBBBA \S 83002 \textbullet{} ED Final Rule Fact Sheet, May 18, 2026 \textbullet{} D\textsuperscript{2} Strategies (Manhattan Inst.), April 2026.}\par}
\end{frame}

% ====================================================================
% 3.  Distinctive opportunity for 4-year + state precedents
% ====================================================================
\frametitleEyebrow{Position / why 4-year universities, and the proof}
\begin{frame}{Workforce Pell was written for institutions like SOU}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.54\textwidth}
      \rmfamily\fontsize{10.5pt}{13.5pt}\selectfont
      \color{textPrimary}
      \eyebrow{the distinctive 4-year market}
      \vspace{4pt}
      \begin{itemize}\itemsep2pt
        \item \textbf{Bachelor's-degree holders are eligible} \textemdash{}
              rare in Pell. Career-changers, alumni, second-degree
              candidates.
        \item \textbf{Countercyclical} to the enrollment cliff:
              traditional 18-year-olds decline; adult re-skilling
              grows with AI \& industry transformation.
        \item \textbf{41.9M Americans} aged 25\textendash 64 have
              some college but no credential (SCNC).
        \item Four-year accreditation, stackable credit, and
              bachelor's-ladder pathways are \textbf{moats that
              for-profit short-course providers cannot cross}.
      \end{itemize}
    \end{column}
    \begin{column}{0.44\textwidth}
      \rmfamily\fontsize{9.6pt}{12.4pt}\selectfont
      \color{textPrimary}
      \eyebrow{states already running this play}
      \vspace{4pt}
      \begin{itemize}\itemsep2pt
        \item \textbf{Virginia} \textemdash{} \$140.8M New Economy
              Workforce Credential Grant: 90\% completion, 67\%
              earn industry credentials, 60\%+ had no prior
              postsecondary.
        \item \textbf{Tennessee} \textemdash{} \$1.5B coordinated
              short-credential-to-degree pathway model.
        \item \textbf{Texas} \textemdash{} \$605.7M
              performance-based community-college funding.
        \item \textbf{Iowa, Louisiana} \textemdash{} robust
              noncredit + wage-record data systems already exist.
        \item \textbf{Oregon} \textemdash{} \textit{state-level data
              \& approval infrastructure not yet built.} SOU can
              be the regional model.
      \end{itemize}
    \end{column}
  \end{columns}
  \vspace{6pt}
  {\centering
   \monoLabel{State precedents \& SCNC figure: D\textsuperscript{2} Strategies, \emph{Workforce Pell Grants: A Strategic Opportunity for Four-Year Institutions}, Manhattan Inst., April 2026.}\par}
\end{frame}

\frametitleEyebrow{Operating model / structurally a public-private partnership}
\begin{frame}{Workforce Pell is, by design, a public-private partnership}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.52\textwidth}
      \rmfamily\fontsize{10.5pt}{14pt}\selectfont
      \color{textPrimary}
      The federal rule does not let an institution run a
      Workforce Pell program alone. Three partnerships are
      \textbf{statutory}:\par
      \vspace{6pt}
      \begin{itemize}\itemsep2pt
        \item \pdot{cPell}\,\textbf{Government.} Governor +
              State Workforce Board certify alignment to
              high-skill, high-wage, in-demand occupations.
              Re-cert every two years.
        \item \pdot{cBiz}\,\textbf{Employers.} May deliver up
              to 25\% of instruction (49\% in Registered
              Apprenticeship). May co-fund tuition. Hiring
              commitments support the 70\% placement test.
        \item \pdot{cSou}\,\textbf{Institution.} Accreditation,
              curriculum, credit, outcomes tracking, advising,
              and academic ladder.
      \end{itemize}
      \vspace{6pt}
      \color{textSecondary}\itshape\fontsize{10pt}{13pt}\selectfont
      ``Treat employers as co-investors in student success.''

\begin{frame}{Each completer is a defensible new revenue line}

      \rmfamily\fontsize{10pt}{13.5pt}\selectfont
      \color{textPrimary}
      \eyebrow{the per-student revenue envelope}
      \vspace{4pt}
      \begin{itemize}\itemsep2pt
        \item Federal cap on tuition + fees =\\
              median completer earnings $-$ \$23{,}475
        \item Avg Pell grant for these programs $\approx$
              \textbf{\$2{,}200/yr} (CBO).
              Employer top-up and other aid bridge the gap.
        \item For a \$60{,}000-median-earnings program:\\
              cap $\approx$ \textbf{\$36{,}525 per student},\\
              well above what most short-course providers charge.
        \item Pell pays first dollar (subject to packaging
              rules). Student carries no debt.
      \end{itemize}
